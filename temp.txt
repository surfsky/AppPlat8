            var user = Auth.GetUser();
            List<CheckTag> tree;
            if (user.Name == "admin")
            {
                tree = CheckTag.GetTree();
            }
            else
            {
                var allTags = CheckTag.IncludeSet.ToList();
                var authOrgId = user.AuthOrgId ?? user.OrgId;

                // 非管理员可见：本组织标签 + 全局标签（OrgId 为空）
                var visible = allTags
                    .Where(t => t.OrgId == null || t.OrgId == authOrgId)
                    .ToList();

                // 若当前可见节点的父节点不可见，补齐祖先，避免树结构断层导致子节点丢失。
                var visibleMap = visible.ToDictionary(t => t.Id, t => t);
                var allMap = allTags.ToDictionary(t => t.Id, t => t);
                var queue = visible.Select(t => t.ParentId).Where(id => id != null).ToList();
                while (queue.Count > 0)
                {
                    var parentId = queue[0];
                    queue.RemoveAt(0);
                    if (parentId == null || visibleMap.ContainsKey(parentId.Value))
                        continue;

                    if (allMap.TryGetValue(parentId.Value, out var parent))
                    {
                        visibleMap[parent.Id] = parent;
                        if (parent.ParentId != null)
                            queue.Add(parent.ParentId);
                    }
                }

                tree = visibleMap.Values.ToList().ToTree();
            }

            if (debug)
            {
                var flat = new List<object>();
                void Walk(IEnumerable<CheckTag> nodes)
                {
                    if (nodes == null)
                        return;
                    foreach (var n in nodes)
                    {
                        flat.Add(new
                        {
                            n.Id,
                            n.Name,
                            n.ParentId,
                            n.OrgId,
                            n.TreeLevel
                        });
                        Walk(n.Children);
                    }
                }

                Walk(tree);
                Console.WriteLine($"[GetCheckTagTree] user={user?.Name}, count={flat.Count}, nodes={flat.ToJson()}");
                return new
                {
                    user = user?.Name,
                    tree,
                    nodes = flat
                }.ToResult();
            }

            return tree.ToResult();
