                var visible = allTags
                    .Where(t => t.OrgId == null || t.OrgId == authOrgId)
                    .ToList();

                // 若当前可见节点的父节点不可见，补齐祖先，避免树结构断层导致子节点丢失。
                var visibleMap = visible.ToDictionary(t => t.Id, t => t);
                var allMap = allTags.ToDictionary(t => t.Id, t => t);
                var queue = visible.Select(t => t.ParentId).Where(id => id != null).ToList();
                while (queue.Count > 0)
                {
                    var parentId = queue[0];
                    queue.RemoveAt(0);
                    if (parentId == null || visibleMap.ContainsKey(parentId.Value))
                        continue;

                    if (allMap.TryGetValue(parentId.Value, out var parent))
                    {
                        visibleMap[parent.Id] = parent;
                        if (parent.ParentId != null)
                            queue.Add(parent.ParentId);
                    }
                }

                tree = visibleMap.Values.ToList().ToTree();
            }

            function parseDataJsonTable() {
                if (isEditMode) return;
                const wrap = document.getElementById('datajson-table-wrap');
                if (!wrap) return;

                const textarea = document.querySelector('textarea[name="item.dataJson"], textarea[name="Item.DataJson"], #item_dataJson, #Item_DataJson');
                const text = (textarea && textarea.value ? textarea.value : '').trim();
                if (!text) {
                    wrap.textContent = '暂无数据';
                    return;
                }

                let obj;
                try {
                    obj = JSON.parse(text);
                } catch {
                    wrap.innerHTML = `<table class="json-kv-table"><tr><th>raw</th><td>${text.replaceAll('<', '&lt;')}</td></tr></table>`;
                    return;
                }

                let rows = [];
                if (Array.isArray(obj)) {
                    rows = obj.map((v, i) => ({ key: `[${i}]`, value: typeof v === 'string' ? v : JSON.stringify(v) }));
                } else if (obj && typeof obj === 'object') {
                    rows = Object.keys(obj).map(k => ({ key: k, value: typeof obj[k] === 'string' ? obj[k] : JSON.stringify(obj[k]) }));
                } else {
                    rows = [{ key: 'value', value: String(obj) }];
                }

                const html = rows.map(r => `<tr><th>${String(r.key).replaceAll('<', '&lt;')}</th><td>${String(r.value).replaceAll('<', '&lt;')}</td></tr>`).join('');
                wrap.innerHTML = `<table class="json-kv-table">${html}</table>`;
            }
            setTimeout(parseDataJsonTable, 200);
