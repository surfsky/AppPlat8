        /// <summary>构建树结构（父子层级）</summary>
        public static List<T> BuildTreeHierarchy<T>(this List<T> items)
            where T : class, ITree<T>
        {
            var dict = items.ToDictionary(x => x.Id);
            var roots = new List<T>();

            // 重置所有节点的子节点列表，防止因 EF Core 自动填充或多次调用导致的重复数据
            foreach (var item in items)
            {
                item.Children = new List<T>();
            }

            foreach (var item in items)
            {
                if (item.ParentId.HasValue && dict.ContainsKey(item.ParentId.Value))
                {
                    var parent = dict[item.ParentId.Value];
                    parent.Children.Add(item);
                    item.Parent = parent;
                }
                else
                {
                    roots.Add(item);
                }
            }
            return roots;
        }
        
        /// <summary>构建树结构</summary>
        static int DoBuildTree<T>(List<T> items, long? rootId, int level, ref List<T> result)
            where T : class, ITree
        {
            int count = 0;
            var root = items.FirstOrDefault(d => d.Id == rootId);
            foreach (var item in items.Where(d => d.ParentId == rootId))
            {
                count++;
                item.TreeLevel = level;
                item.IsTreeLeaf = true;
                item.Enabled = true;
                //if (root?.Children != null)
                //    root.Children.Add(item);
                result.Add(item);

                level++;
                // 如果这个节点下没有子节点，则这是个终结节点
                int childCount = DoBuildTree(items, item.Id, level, ref result);
                if (childCount != 0)
                    item.IsTreeLeaf = false;
                level--;
            }
            return count;
        }

        /// <summary>拷贝树结构列表（并构造父子结构）</summary>
        public static List<T> CloneTree<T>(this List<T> items)
            where T : class, ITree<T>
        {
            // 先拷贝一份
            var result = new List<T>();
            foreach (var item in items)
                result.Add((T)item.Clone());

            // 构建父子结构
            foreach (var item in result)
            {
                item.Parent = result.FirstOrDefault(t => t.Id == item.ParentId);
                item.Children = result.Where(t => t.ParentId == item.Id).ToList();
            }

            //
            result = result.BuildTreeHierarchy();
            return result;
        }

        /// <summary>构建树节点</summary>
        /// <param name="lookup">组织lookup</param>
        /// <param name="parentId">父ID</param>
        /// <returns></returns>
        public static List<TreeItem> BuildTree(ILookup<long?, ITree> lookup, long? parentId)
        {
            var nodes = new List<TreeItem>();
            foreach (var item in lookup[parentId])
            {
                var children = BuildTree(lookup, item.Id);
                var node = new TreeItem
                {
                    Value = item.Id,
                    Label = item.Name,
                    Children = children.Any() ? children : null
                };
                nodes.Add(node);
            }

            return nodes;
        }

        /// <summary>检索</summary>
        public static IQueryable<RoleUser> Search(
            long? userId = null,
            string userName = null,
            long? roleId = null,
            string roleName = null
            )
        {
            IQueryable<RoleUser>       q = Set.Include(t => t.User).Include(t => t.Role);
            if (userId.IsNotEmpty())   q = q.Where(t => t.UserId == userId);
            if (userName.IsNotEmpty()) q = q.Where(t => t.User.Name.Contains(userName));
            if (roleId.IsNotEmpty())   q = q.Where(t => t.RoleId == roleId);
            if (roleName.IsNotEmpty()) q = q.Where(t => t.Role.Name.Contains(roleName));
            return q;
        }
